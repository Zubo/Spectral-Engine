#version 330 core
out vec4 FragColor;

in vec3 Normal;
in vec3 FragPos;

uniform vec3 objectColor;
uniform vec3 lightColor;
uniform vec3 lightPos;

void main()
{
	vec3 lightDir = (lightPos - FragPos);
	float diffuseIntensity = max(0, dot(normalize(Normal), normalize(lightDir)));
	vec3 diffuseLight = diffuseIntensity * lightColor;
	float ambientStrength = 0.2F;
	vec3 light = (diffuseLight  + ambientStrength) * objectColor;
	FragColor = vec4(light, 1.0F);
}
